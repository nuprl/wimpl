wassail 
  wassail callgraph ../callgraph.wasm wassail.dot 
  dot -Tps wassail.dot -o wassail.ps

aWsm (wasm -> llvm) 
  awsm ../callgraph.wasm -o callgraph.bc // gives you foo.bc file  
  llvm-dis-12 callgraph.bc -o callgraph.ll // get human readable format 
  opt-12 callgraph.bc --dot-callgraph //get dot file 
  dot -Tps callgraph.bc.callgraph.dot -o awsm.ps
  
wavm 
   wavm compile --format=unoptimized-llvmir callgraph.wasm wavm.ll //get .ll file 
   llvm-as-12 wavm.ll -o wavm.bc 
   opt-12 wavm.bc --dot-callgraph //get dot file 
   dot -Tps wavm.bc.callgraph.dot -o wavm.ps //get ps file

  twiggy paths ../callgraph.wasm > paths.csv 
  python3 get-callgraph.py paths.csv
